Extract CSV to Excel







packages = ["pandas", "openpyxl"] import pandas as pd from pyodide.ffi import to_js from js import document, Blob, URL def process_csv(event=None): status = document.getElementById("status") file_input = document.getElementById("csv-file") pass_type = document.getElementById("pass-type").value output_name = document.getElementById("output-name").value or "output.xlsx" if not file_input.files.length: status.innerText = "Please upload a CSV file." return file = file_input.files.item(0) def onload(evt): try: csv_text = evt.target.result df = pd.read_csv(pd.compat.StringIO(csv_text)) available_columns = list(df.columns) if pass_type == "1": selected_indexes = [32, 8, 9, 1, 3, 7] else: selected_indexes = [17, 1, 3, 7, 15, 13, 57, 20, 43] if any(idx < 1 or idx > len(available_columns) for idx in selected_indexes): status.innerText = "One or more requested column positions are out of range for this CSV." return columns = [available_columns[idx-1] for idx in selected_indexes] missing = [col for col in columns if col not in df.columns] if missing: status.innerText = f"These columns were not found: {', '.join(missing)}" return # Write to Excel in memory import io output = io.BytesIO() df.loc[:, columns].to_excel(output, index=False) output.seek(0) # Download blob = Blob.new([output.getvalue()], {"type": "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet"}) url = URL.createObjectURL(blob) a = document.createElement("a") a.href = url a.download = output_name a.click() URL.revokeObjectURL(url) status.innerText = f"Excel file '{output_name}' generated and downloaded." except Exception as e: status.innerText = f"Error: {e}" reader = __new__(js.FileReader()) reader.onload = onload reader.readAsText(file)
